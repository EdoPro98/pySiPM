7 def addDCR(rate):
8  """!@brief Generation of dark count events."""
9  """!
10  Function that generates times for dark count events (DCR).
11  Events are generated as an uniform poisson process, hence the time
12  distance between consecutive DCR events follows an exponential
13  distribution with a mean value of: @f$\tau=\frac{1}{DCR}@f$
14  @param rate: Dark counts rate un Hz.
15  @return dcrTime: List containing times of DCR events in ns.
16  @return nDcr: Number of DCR events generated
17  """
18 
19  dcrTime = []
20  last = 0
21  # Generate using exponential distribution of delays
22  while last < SIGLEN:
23  last = frandom.exponential(1e9 / rate, 1).item()
24  dcrTime.append(last)
25 
26  # Delete last one (exceeding SIGLEN)
27  dcrTime.pop(-1)
28  nDcr = len(dcrTime)
29  return dcrTime, nDcr

48 def addXT(times, idx, xt):
49  """!@brief Generation of optical crosstalk events."""
50  """!
51  Function that generates optical crosstalk events(XT). Each photoelectron
52  has generates a poissonian number of XT events accordingly to the crosstalk
53  probability. XT events are generated in the 8 neighbouring cells and have
54  the same time of the main cell.@n
55  XT events can also generate other XT events in theyr neighbouring cells.
56  @param times: List containing the time of each SiPM cell avalanche,
57  including DCR events.
58  @param idx: List containing the ID of each fired cell.
59  @param xt: Value (probability) of XT events.
60  @return times: List containing the time at which SiPM cells are fired
61  including XT events
62  @return idx: List containing the ID of the hitted cells including
63  XT events.
64  @return nXt: Number of Xt events generated.
65  """
66  neighbour = (1, -1, CELLSIDE, -CELLSIDE, 1+CELLSIDE, 1-CELLSIDE,
67  -1+CELLSIDE, -1-CELLSIDE)
68  npe = len(times)
69  nXt = 0
70  i = 0
71 
72  while i < npe:
73  nxt = frandom.poisson(xt, 1).item()
74  nXt += nxt
75  npe += nxt
76  for j in range(nxt):
77  idx.append(idx[i] + random.choice(neighbour))
78  times.append(times[i])
79  i += 1
80 
81  return times, idx, nXt

84 def addAP(times, h, ap):
85  """!@brief Generation of afterpulses."""
86  """!
87  Function that generates afterpulses(AP) events and adds them to the list of
88  events to simulate. Each avalanche generates a poissonian number of AP
89  events. APs are delayed from teyr main signal following a fast and slow
90  exponential distributions. Theyr relative signal height is calculated
91  consdering the cell recovery time as an RC circuit:
92  @f[h=1-e^{-\frac{\Delta_t}{\tau}}@f]
93  @param times: List containing the time at which SiPM cells are fired,
94  including DCR and XT events.
95  @param h: List containing the relative signal height of each
96  fired cell.
97  @param ap: Value (probability) of AP events.
98  @return times: List containing the time at which SiPM cells are fired,
99  including AP events.
100  @return h: List containing the relative signal height of each
101  hitted cell.
102  @return nAp: Number of Ap events generated.
103  """
104  npe = len(times)
105  nAp = 0
106  i = 0
107 
108  while i < npe:
109  nap = frandom.poisson(ap, 1).item()
110  if nap:
111  delays = frandom.exponential(TAUAPFAST, nap) + frandom.exponential(TAUAPSLOW, nap)
112  heights = 1 - exp(-delays / CELLRECOVERY)
113  aptimes = times[i] + delays
114 
115  mask = aptimes < SIGLEN
116  times.extend(aptimes[mask])
117  h.extend(heights[mask])
118 
119  # Count only ap inside time window
120  npe += np.count_nonzero(mask)
121  nAp += np.count_nonzero(mask)
122  i += 1
123 
124  return times, h, nAp

127 def SiPMEventAction(times, idx):
128  """!@brief Calculates relative signal height for each photoelectron."""
129  """!
130  Since a SiPM cell may be hitted multiple times, recovery time has to be
131  considered.
132  If all cell IDs are different all signal heights are left unchanged with a
133  value of 1, else if an ID appears multiple times then the first (in time)
134  hit gives a signal height of 1 and the following will have a lower relative
135  height.
136  The relative signal height, after a @f$\Delta_t@f$ time from the previous
137  hit, is calculated supposing that each SiPM cell recovers as an RC circuit:
138  @f[h(\Delta_t)=1-e^{-\frac{t}{\tau}}@f]
139  @param times: List containing the time of all events that have generated
140  an avalanche in the SiPM cells.
141  @param idx: List containing the corresponding cell IDs of the events.
142  @return h: List containing the relative signal height of each
143  avalanche.
144  """
145 
146  # Converting to Numpy arrays
147  times = np.asarray(times, dtype=np.float32)
148  times = times[times < SIGLEN]
149  # Start with all signals with height 1
150  h = np.ones_like(times)
151 
152  # If no repeated hits skip
153  if len(idx) != len(set(idx)):
154  idx = np.asarray(idx, dtype=np.int32)
155 
156  # Sorting times and theyr idx
157  idxsort = np.argsort(times)
158  times = times[idxsort]
159  idx = idx[idxsort]
160 
161  _, uniqindex, uniqcts = np.unique(idx, return_index=True,
162  return_counts=True)
163 
164  # Idx of cells fired more than once
165  midx = uniqindex[uniqcts > 1]
166 
167  for m in midx:
168  mtimes = times[idx == idx[m]]
169  htemp = 1 - exp(-np.diff(mtimes, prepend=0) / CELLRECOVERY)
170  htemp[0] = 1
171  h[idx == idx[m]] = htemp
172 
173  return times.tolist(), h.tolist()

245 def signalAnalysis(signal, intstart, intgate, threshold):
246  """!@brief Features extraction from signals."""
247  """!
248  Function that extracts some simple features from signals considering an
249  integration gate. At the moment the features considered are:
250  - Peak: signal peak in the integration gate.
251  - Integral: sum of samples in the integration gate corrected by the
252  sampling time.
253  - Time of Arrival: time position of the first sample above the
254  threshold.
255  - Time over Threshold: Number fo samples above the threshold corrected
256  by the sampling time.
257  - Time of Peak: time position of the sample in the peak.
258  This list might be expanded in future with more complex features.
259  @param signal: Array containing the digitized signal.
260  @param intstart: Integer value containing the index corresponding to the
261  start of the integration gate.
262  @param intgate: Integer value containing the lenght of the integration
263  gate in units of samples.
264  @param threshold Threshold used to calculate the values described above.
265  It is automatically set to 1.5 times the single
266  photoelectron peak height.
267  @return Returns the features extracted. If thesignal is below
268  the threshold all features are set to -1.
269  """
270 
271  sigingate = signal[intstart:intstart + intgate]
272 
273  test = sigingate.max()
274 
275  if test > threshold:
276  mask = sigingate > threshold
277  peak = test
278  integral = sigingate.sum() * SAMPLING
279  toa = mask.argmax() * SAMPLING
280  tot = np.count_nonzero(mask) * SAMPLING
281  top = sigingate.argmax() * SAMPLING
282  else:
283  peak = -1
284  integral = -1
285  toa = -1
286  tot = -1
287  top = -1
288 
289  return peak, integral, toa, tot, top
