18  """!@brief Function that generates the signal from all SiPM cells at once."""
19  """!
20  This is the "full" version that computes the signal shape on GPU. The
21  signals are generated at once using a CUDA kernel that generates the
22  signals of each cell in a matrix (one cell per row) and then sums everything
23  column-wise obtaining the complete SiPM signal.
24 
25  @param t: Array containing times at which each the cell is triggered
26  @param h: Array containing the relative pulse height of each cell signal
27 
28  @return s: Array containing the generated SiPM signal
29  """
30  # Number of signals to generate
31  n = t.size
32  # Generate matrix containing times of each fired cell
33  vect = (cp.arange(SIGPTS, dtype=cp.int32) + cp.zeros((n, 1), dtype=cp.int32) - t[:, None])
34  # Zero before the signal
35  vect[vect < 0] = 0
36  # Transpose array of height values
37  h = h[:, None] * PEAKRATIO
38  # Call kernel to generate singal
39  sig = cp.sum(signalShapeGPU(vect, TFALL / SAMPLING, TRISE / SAMPLING, h), axis=0)
40  # If there are afterpulses generate theyr signals

46  """! @brief Generation of full SiPM signal."""
47  """!
48  Function that generates the full SiPM signal as the sum of the signals
49  of each cell starting from gaussian noise.
50  If there are fewer hitted cells the signal is generated on CPU to avoid
51  the overhead caused by moving data to GPU memory. Also if there are too
52  many hitted cells the signal is generated on CPU to avoid an out-of-memory
53  condition since usually VRAM is smaller than system memory.
54  Those conditions are controlled by @ref variables.CPUTHRESHOLD and
55  @ref variables.GPUMAX and are set by default at 100 and 2000.
56 
57  @param times: List containing the time at wich SiPM cells are fired,
58  including DCR, XT and AP events
59  @param sigH: List containing the correspondin relative pulse height
60  of each fired cell.
61  @param snr: Signal to noise ratio converted into the RMS of the
62  gaussian noise.
63  @param basespread: Sigma of the value to add as baseline spread.
64 
65  @return signal: Array containing the complete sigitized SiPM signal.
66  """
67  times = np.array(times, dtype=np.float32, copy=False)
68  sigH = np.array(sigH, dtype=np.float32, copy=False)
69  sigH = sigH[times < SIGLEN]
70  times = (times[times < SIGLEN] / SAMPLING).astype(np.uint32)
71  signal = normal(0, SNR, SIGPTS)
72  sigH = sigH * normal(1, CCGV, times.size) # Each signal has a ccgv
73  if (times.size < CPUTHRESHOLD) or (times.size > GPUMAX):
74  for i in range(times.size):
75  signal += PulseCPU(times[i], sigH[i])
76  return signal
77  else:
78  signal = cp.asarray(signal)
79  signal += PulseGPU(cp.asarray(times), cp.asarray(sigH))
80  return cp.asnumpy(signal)
