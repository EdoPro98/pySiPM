184  endif

187  right = n
188  temp = array(2)
189 
190  do i = 2,n - 1
191  if (temp .lt. p) then
192  do j = left, 1, -1
193  if (array(j) .le. temp) exit
194  array(j + 1) = array(j)
195  end do
196  array(j + 1) = temp
197  temp = array(left + 2)
198  left = left + 1
199  else
200  next = array(right - 1)
201  do j = right, n
202  if (array(j) .ge. temp) exit
203  array(j - 1) = array(j)
204  end do
205  array(j - 1) = temp
206  temp = next
207  right = right - 1
208  endif
209  end do
